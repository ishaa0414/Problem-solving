Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
        {
        let hourangle=(30*h)+(0.5*m);
        let minangle=6*m;
        let minimalangle=0;
        if(h===12 && m===0){
          return 0;
        }
        if(hourangle > minangle){
          minimalangle=(hourangle-minangle);
        }
        else{
          minimalangle=(minangle-hourangle);
        }
        return minimalangle;
        };

Perfect Number Check.

        var Perfect_Check = (N) => 
        {
          let num=0;
          for(let i=0;i <= N;i++){
            if(N%i === 0){
              num=num+i;
            }
          }
            if(num===N){
              return "YES";
            }    
            else{
              return "NO";
            }
        };

Reverse a Number.

        var Reverse_Number = (N) => 
        {
          let a=0;
          let result=0;
         while(N>0){
            a = N%10;
            result = result*10 + a;
          N = Math.floor(N/10);
         }
         return result;
        };
